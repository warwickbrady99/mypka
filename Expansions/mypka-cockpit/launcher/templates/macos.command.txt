REFERENCE TEMPLATE — macOS launcher (.txt on purpose; NOT runnable as-is).
The installing LLM adapts the values below to the user's real cockpit path / port
and writes the result as `start-cockpit.command` in the cockpit root, then
`chmod +x` it. See ../GENERATE-LAUNCHER.md. Ships as text so nothing pre-made runs.
================================================================================
#!/bin/bash
# start-cockpit.command — myPKA Cockpit launcher (macOS). Double-click to run.
# Generated locally from launcher/templates/macos.command.txt — review before use.
set -euo pipefail

# --- config (the LLM fills these in for THIS machine) ------------------------
PORT="${PORT:-4317}"
# MYPKA_ROOT: leave unset for auto-detection (server/repoRoot.js finds the
# scaffold root). Uncomment + set ONLY for a non-standard layout:
# export MYPKA_ROOT="/absolute/path/to/your/myPKA"

# --- 1. resolve the cockpit dir and cd into it -------------------------------
# The .command file lives in the cockpit root; cd to its own directory so all
# relative paths (server/, web/, scripts/) and `npm --prefix web` resolve.
COCKPIT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
cd "$COCKPIT_DIR"

# Resolve the scaffold-root mypka.db the server will open. The standard layout is
# <root>/Expansions/mypka-cockpit/, so the root is three levels up. If the user
# set MYPKA_ROOT (non-standard layout), honor it — the server reads the same var.
if [ -n "${MYPKA_ROOT:-}" ]; then
  DB_PATH="$MYPKA_ROOT/mypka.db"
else
  DB_PATH="$(cd "$COCKPIT_DIR/../.." && pwd)/mypka.db"
fi

# --- 2. ensure mypka.db exists (the server REQUIRES it; absence = dead cockpit)
# The server opens mypka.db read-only and HARD-EXITS if it is missing, so the
# launcher must GUARANTEE a DB with the core schema before starting node. We
# check first (fast path: a normal start never rebuilds) and only create when the
# DB is missing or coreless.
db_has_core() {  # 0 = DB exists AND carries the core `journal` table
  python3 - "$DB_PATH" <<'PY' 2>/dev/null
import os, sqlite3, sys
p = sys.argv[1]
if not os.path.isfile(p): sys.exit(1)
try:
    c = sqlite3.connect(f"file:{p}?mode=ro", uri=True)
    sys.exit(0 if c.execute(
        "SELECT 1 FROM sqlite_master WHERE type='table' AND name='journal'"
    ).fetchone() else 1)
except sqlite3.Error:
    sys.exit(1)
PY
}

if db_has_core; then
  # FAST PATH — the DB is already here. Refresh it from markdown if Python+PyYAML
  # are available (non-destructive); otherwise just serve the existing DB.
  if command -v python3 >/dev/null 2>&1 && python3 -c "import yaml" >/dev/null 2>&1; then
    echo "Refreshing mypka.db from your markdown (non-destructive)…"
    python3 "scripts/regen-mypka-db.py" || echo "  (regen failed — using existing mypka.db)"
  else
    echo "Python 3 + PyYAML not found — skipping DB refresh (existing mypka.db will serve)."
    echo "  To enable refreshes: install Python 3, then  pip3 install --user pyyaml"
  fi
else
  # FIRST-RUN PATH — no usable DB yet. Build it (core schema + every cockpit
  # module) via the idempotent installer, which auto-bootstraps the base DB.
  echo "No mypka.db yet — creating it (core schema + all cockpit modules)…"
  if ! command -v python3 >/dev/null 2>&1 || ! python3 -c "import yaml" >/dev/null 2>&1; then
    echo ""
    echo "  Cannot create mypka.db: Python 3 with the PyYAML package is required"
    echo "  to read your markdown the first time. Install them, then re-run me:"
    echo ""
    echo "      • Python 3:  https://www.python.org/downloads/  (or:  brew install python)"
    echo "      • PyYAML:    pip3 install --user pyyaml"
    echo ""
    echo "  (Once mypka.db exists, future launches do NOT need Python.)"
    echo "  Press any key to close." ; read -r -n 1 _ || true
    exit 1
  fi
  if ! python3 "sqlite-extension/install-extensions.py" "$DB_PATH" --all; then
    echo ""
    echo "  Could not create mypka.db (see the message above). The cockpit cannot"
    echo "  start without it. Fix the issue, then re-run me."
    echo "  Press any key to close." ; read -r -n 1 _ || true
    exit 1
  fi
fi

# --- 3. first-run install + build (skipped on later launches) ----------------
[ -d node_modules ]      || { echo "Installing server deps…"; npm install --no-audit --no-fund; }
[ -d web/node_modules ]  || { echo "Installing web deps…";    npm --prefix web install --no-audit --no-fund; }
[ -d web/dist ]          || { echo "Building the web app…";   npm --prefix web run build; }

# --- 4. free the port (kill a stale instance; no-op if nothing is listening) -
if lsof -ti "tcp:$PORT" >/dev/null 2>&1; then
  echo "Port $PORT busy — stopping the previous instance…"
  lsof -ti "tcp:$PORT" | xargs kill 2>/dev/null || true
  sleep 1
fi

# --- 5. open the browser (SPA retries until the API is up) -------------------
open "http://127.0.0.1:$PORT/" || true

# --- 6. start the server, loopback-only, with the launcher defaults ----------
# WORKBENCH_WRITE_ENABLED / PLAN_WRITE_ENABLED default to 1 here (set 0 for a
# fully read-only cockpit). MYPKA_ROOT is exported above only if the user set it.
# NODE_ENV=production keeps Express from serving its default HTML stack trace on
# an unhandled error (which would leak absolute server paths); paired with the
# JSON error handler in server.js.
echo "Starting the cockpit on http://127.0.0.1:$PORT/  (close this window to stop it)"
NODE_ENV=production PORT="$PORT" WORKBENCH_WRITE_ENABLED=1 PLAN_WRITE_ENABLED=1 exec node server/server.js
